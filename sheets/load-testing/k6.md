# k6 (Modern Load Testing)

JavaScript-based load testing tool with built-in support for HTTP, WebSocket, gRPC, virtual users, thresholds, checks, scenarios, and cloud execution for performance testing APIs and web applications.

## Installation

```bash
# Linux
apt install k6                            # Debian/Ubuntu (via k6 repo)
snap install k6                           # Snap

# macOS
brew install k6

# Docker
docker run --rm -i grafana/k6 run - <script.js

# Verify
k6 version
```

## Basic Script

```javascript
// script.js
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
  vus: 10,                                // 10 virtual users
  duration: '30s',                        // run for 30 seconds
};

export default function () {
  const res = http.get('https://test-api.k6.io/public/crocodiles/');

  check(res, {
    'status is 200': (r) => r.status === 200,
    'response time < 500ms': (r) => r.timings.duration < 500,
  });

  sleep(1);                               // think time between iterations
}
```

```bash
# Run the script
k6 run script.js
```

## HTTP Methods

```javascript
import http from 'k6/http';

// GET
const res = http.get('https://api.example.com/users');

// POST JSON
const payload = JSON.stringify({ name: 'Alice', email: 'alice@example.com' });
const params = { headers: { 'Content-Type': 'application/json' } };
const res = http.post('https://api.example.com/users', payload, params);

// PUT
http.put('https://api.example.com/users/1', payload, params);

// PATCH
http.patch('https://api.example.com/users/1', payload, params);

// DELETE
http.del('https://api.example.com/users/1');

// Batch requests (parallel)
const responses = http.batch([
  ['GET', 'https://api.example.com/users'],
  ['GET', 'https://api.example.com/posts'],
]);
```

## Checks and Thresholds

```javascript
import http from 'k6/http';
import { check } from 'k6';

export const options = {
  thresholds: {
    http_req_duration: ['p(95)<500'],      // 95th percentile < 500ms
    http_req_failed: ['rate<0.01'],         // error rate < 1%
    checks: ['rate>0.99'],                  // 99% of checks pass
    http_reqs: ['rate>100'],                // at least 100 RPS
  },
};

export default function () {
  const res = http.get('https://api.example.com/health');

  check(res, {
    'status is 200': (r) => r.status === 200,
    'body contains ok': (r) => r.body.includes('ok'),
    'response time OK': (r) => r.timings.duration < 200,
    'content-type is json': (r) => r.headers['Content-Type'] === 'application/json',
  });
}
```

## Scenarios

```javascript
export const options = {
  scenarios: {
    // Constant VUs
    constant_load: {
      executor: 'constant-vus',
      vus: 50,
      duration: '5m',
    },

    // Ramping VUs (load profile)
    ramp_up: {
      executor: 'ramping-vus',
      startVUs: 0,
      stages: [
        { duration: '2m', target: 50 },    // ramp up to 50
        { duration: '5m', target: 50 },    // hold at 50
        { duration: '2m', target: 100 },   // ramp to 100
        { duration: '5m', target: 100 },   // hold at 100
        { duration: '2m', target: 0 },     // ramp down
      ],
    },

    // Constant arrival rate (open model)
    constant_rate: {
      executor: 'constant-arrival-rate',
      rate: 200,                           // 200 iterations per timeUnit
      timeUnit: '1s',                      // = 200 RPS
      duration: '5m',
      preAllocatedVUs: 50,
      maxVUs: 200,
    },

    // Ramping arrival rate
    ramping_rate: {
      executor: 'ramping-arrival-rate',
      startRate: 10,
      timeUnit: '1s',
      preAllocatedVUs: 50,
      stages: [
        { duration: '2m', target: 100 },
        { duration: '5m', target: 100 },
        { duration: '2m', target: 200 },
      ],
    },

    // Shared iterations (fixed total)
    smoke: {
      executor: 'shared-iterations',
      vus: 5,
      iterations: 100,
    },
  },
};
```

## Data Parameterization

```javascript
import { SharedArray } from 'k6/data';
import papaparse from 'https://jslib.k6.io/papaparse/5.1.1/index.js';
import { open } from 'k6/experimental/fs';

// JSON data
const users = new SharedArray('users', function () {
  return JSON.parse(open('./users.json'));
});

// CSV data
const csvData = new SharedArray('csv', function () {
  return papaparse.parse(open('./data.csv'), { header: true }).data;
});

export default function () {
  const user = users[Math.floor(Math.random() * users.length)];
  http.post('https://api.example.com/login', JSON.stringify({
    username: user.username,
    password: user.password,
  }));
}
```

## CLI Options

```bash
# Override VUs and duration
k6 run --vus 100 --duration 60s script.js

# Run with iterations limit
k6 run --iterations 1000 script.js

# Set environment variables
k6 run -e BASE_URL=https://staging.example.com script.js

# HTTP debug (print request/response)
k6 run --http-debug script.js

# Quiet mode (no progress bar)
k6 run --quiet script.js

# Summary export
k6 run --summary-export=summary.json script.js

# Abort on threshold failure
k6 run --abort-on-tainted-threshold script.js
```

## WebSocket Testing

```javascript
import ws from 'k6/ws';
import { check } from 'k6';

export default function () {
  const res = ws.connect('wss://echo.websocket.org', {}, function (socket) {
    socket.on('open', () => {
      socket.send('hello');
      socket.setInterval(() => socket.send('ping'), 1000);
    });

    socket.on('message', (msg) => {
      check(msg, { 'message received': (m) => m === 'hello' });
    });

    socket.setTimeout(() => socket.close(), 5000);
  });

  check(res, { 'connected successfully': (r) => r && r.status === 101 });
}
```

## Tips

- Use `constant-arrival-rate` executor instead of `constant-vus` when you need to guarantee a specific request rate regardless of response times.
- Place `sleep()` calls between iterations to simulate realistic user think time and avoid overwhelming targets with zero-delay loops.
- Use `SharedArray` for large datasets to share memory across VUs instead of each VU loading its own copy.
- Set thresholds as exit criteria in CI/CD pipelines so load tests fail the build when performance degrades.
- Start with a smoke test (5 VUs, 1 minute) before scaling up to catch script errors cheaply.
- Use `group()` to organize requests into logical transactions for clearer reporting and per-group metrics.
- Tag requests with `{ tags: { name: 'login' } }` to distinguish endpoints in threshold definitions.
- Use `--http-debug` to troubleshoot unexpected 4xx/5xx responses before scaling up virtual users.
- Export results to InfluxDB + Grafana for real-time dashboards during extended test runs.
- Use `ramping-vus` stages to simulate realistic traffic patterns: ramp up, steady state, spike, ramp down.
- The `k6 cloud` command uploads scripts to Grafana Cloud for distributed execution from multiple geographic regions.
- Avoid file I/O in the default function. Load data in `init` context or use `SharedArray` for test data.
